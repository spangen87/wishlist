**Part A — Create src/app/add-child/page.tsx:**

```typescript
'use client';
import { useEffect } from 'react';
import { useRouter } from 'next/navigation';
import { useAuth } from '@/components/AuthProvider';
import { ChildAccountForm } from '@/components/onboarding/ChildAccountForm';

export default function AddChildPage() {
  const router = useRouter();
  const { user, role, loading } = useAuth();

  // Auth gate — viewer only (same as /onboarding)
  useEffect(() => {
    if (!loading && !user) router.push('/login');
    if (!loading && user && role === 'child') router.push('/wishlist');
  }, [loading, user, role, router]);

  if (loading || !user) {
    return (
      <main className="flex min-h-screen items-center justify-center bg-[#FFF9F5]">
        <p className="text-[#6B7280]">Laddar…</p>
      </main>
    );
  }

  return (
    <main className="flex min-h-screen items-center justify-center p-4 bg-[#FFF9F5]">
      <div className="w-full max-w-sm">
        <h1 className="text-2xl font-bold mb-6 text-center text-[#171717]">
          Lägg till barn
        </h1>
        <ChildAccountForm
          onSuccess={() => router.push('/dashboard')}
        />
        <p className="mt-4 text-sm text-center">
          <a
            href="/dashboard"
            className="text-[#6B7280] underline"
          >
            Tillbaka till instrumentpanelen
          </a>
        </p>
      </div>
    </main>
  );
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → /add-child page | Viewer session required; page renders ChildAccountForm which calls register-child API |
| Firestore users/{uid}.displayName → dashboard card | displayName from Firestore is rendered as text in React — no sanitization needed |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Elevation of Privilege | /add-child page — child role access | mitigate | Auth-gate useEffect redirects role==='child' → /wishlist; child cannot create other child accounts |
| T-03-02 | Information Disclosure | /add-child unauthenticated access | mitigate | Auth-gate useEffect redirects !user → /login before any content renders |
| T-03-03 | Tampering | dashboard displayName from Firestore | accept | displayName rendered as React text node (not innerHTML) — XSS not possible; data stored by Admin SDK via register-child |
| T-03-04 | Spoofing | register page — no auth change | accept | /register page changes are text/style only; all Firebase Auth logic unchanged; security boundary not moved |
</threat_model>
