```typescript
import 'server-only';
import { NextRequest, NextResponse } from 'next/server';
import { adminAuth, adminDb } from '@/lib/firebase/admin';

export async function DELETE(
  request: NextRequest,
  { params }: { params: Promise<{ wishlistId: string }> }
) {
  const { wishlistId } = await params;
  const body = await request.json().catch(() => ({}));
  const { idToken } = body as { idToken?: string };

  if (!idToken) {
    return NextResponse.json({ error: 'idToken required' }, { status: 400 });
  }

  let decoded;
  try {
    decoded = await adminAuth.verifyIdToken(idToken);
  } catch {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  const wishlistSnap = await adminDb.collection('wishlists').doc(wishlistId).get();
  if (!wishlistSnap.exists) {
    return NextResponse.json({ error: 'Not found' }, { status: 404 });
  }

  const data = wishlistSnap.data()!;
  const callerIsOwner = data.childUid === decoded.uid;
  const callerIsParent = (data.parentUids ?? []).includes(decoded.uid);
  if (!callerIsOwner && !callerIsParent) {
    return NextResponse.json({ error: 'Forbidden' }, { status: 403 });
  }

  // 1. Cascade-delete wishlist root + items/* + purchaseStatus/* + activityLog/*
  //    MUST use Admin SDK recursiveDelete — client deleteDoc() leaves subcollections orphaned
  await adminDb.recursiveDelete(adminDb.collection('wishlists').doc(wishlistId));

  // 2. Clean up invite tokens referencing this wishlist (they'd 404 on redeem anyway but take space)
  const inviteSnap = await adminDb.collection('invites')
    .where('wishlistId', '==', wishlistId).get();
  if (!inviteSnap.empty) {
    const inviteBatch = adminDb.batch();
    inviteSnap.docs.forEach((d) => inviteBatch.delete(d.ref));
    await inviteBatch.commit();
  }

  return NextResponse.json({ ok: true });
}
```

Create the directory path `src/app/api/auth/user/[uid]/` and the file `route.ts` inside it.

The route handles two distinct flows based on the target user's `role` field:

**Flow A — child role:** Only a parent in the child's `parentUids` array may trigger this. Sequence: recursiveDelete wishlist → batch-delete users/{uid} + usernames/{username} → cleanup invite tokens → deleteUser (idempotent).

**Flow B — parent or viewer role:** Only the user themselves (`decoded.uid === targetUid`) may trigger this. Sequence: arrayRemove their UID from all wishlists' parentUids + viewerUids → delete users/{uid} → deleteUser (idempotent).

**Critical ordering (Pitfall 2):** Always delete Firestore data BEFORE calling `adminAuth.deleteUser()`. If Firestore succeeds and Auth delete fails, the user can retry. The reverse (Auth gone, Firestore remains) creates a permanent orphan.

Implement exactly:

```typescript
import 'server-only';
import { NextRequest, NextResponse } from 'next/server';
import { adminAuth, adminDb } from '@/lib/firebase/admin';
import { FieldValue } from 'firebase-admin/firestore';

export async function DELETE(
  request: NextRequest,
  { params }: { params: Promise<{ uid: string }> }
) {
  const { uid: targetUid } = await params;
  const body = await request.json().catch(() => ({}));
  const { idToken } = body as { idToken?: string };

  if (!idToken) {
    return NextResponse.json({ error: 'idToken required' }, { status: 400 });
  }

  let decoded;
  try {
    decoded = await adminAuth.verifyIdToken(idToken);
  } catch {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  // Load target user profile to determine role + username
  const userSnap = await adminDb.collection('users').doc(targetUid).get();
  if (!userSnap.exists) {
    return NextResponse.json({ error: 'User not found' }, { status: 404 });
  }
  const userData = userSnap.data()!;
  const role: string = userData.role;
  const username: string | undefined = userData.username;

  if (role === 'child') {
    // Only a parent of this child (in parentUids) may delete the child account
    const wishlistSnap = await adminDb.collection('wishlists').doc(targetUid).get();
    const parentUids: string[] = wishlistSnap.exists
      ? (wishlistSnap.data()!.parentUids ?? [])
      : [];
    if (!parentUids.includes(decoded.uid)) {
      return NextResponse.json({ error: 'Forbidden' }, { status: 403 });
    }

    // Delete Firestore data first (Pitfall 2: Firestore before Auth)
    // 1. Cascade-delete wishlist + items/* + purchaseStatus/* + activityLog/*
    if (wishlistSnap.exists) {
      await adminDb.recursiveDelete(adminDb.collection('wishlists').doc(targetUid));
    }

    // 2. Batch-delete users/{uid} and usernames/{username}
    const batch = adminDb.batch();
    batch.delete(adminDb.collection('users').doc(targetUid));
    if (username) {
      batch.delete(adminDb.collection('usernames').doc(username));
    }
    await batch.commit();

    // 3. Clean up orphaned invite tokens for this wishlist
    const inviteSnap = await adminDb.collection('invites')
      .where('wishlistId', '==', targetUid).get();
    if (!inviteSnap.empty) {
      const inviteBatch = adminDb.batch();
      inviteSnap.docs.forEach((d) => inviteBatch.delete(d.ref));
      await inviteBatch.commit();
    }
  } else {
    // parent or viewer: only the user themselves may delete their own account
    if (decoded.uid !== targetUid) {
      return NextResponse.json({ error: 'Forbidden' }, { status: 403 });
    }

    // Remove UID from all wishlists where they appear as parent or viewer
    const [parentLists, viewerLists] = await Promise.all([
      adminDb.collection('wishlists').where('parentUids', 'array-contains', targetUid).get(),
      adminDb.collection('wishlists').where('viewerUids', 'array-contains', targetUid).get(),
    ]);

    const removalBatch = adminDb.batch();
    parentLists.docs.forEach((d) =>
      removalBatch.update(d.ref, { parentUids: FieldValue.arrayRemove(targetUid) })
    );
    viewerLists.docs.forEach((d) =>
      removalBatch.update(d.ref, { viewerUids: FieldValue.arrayRemove(targetUid) })
    );
    removalBatch.delete(adminDb.collection('users').doc(targetUid));
    await removalBatch.commit();
  }

  // Delete Firebase Auth user — idempotent (Pitfall 3: handle auth/user-not-found)
  try {
    await adminAuth.deleteUser(targetUid);
  } catch (err: unknown) {
    if ((err as { code?: string }).code !== 'auth/user-not-found') throw err;
    // Already deleted from Auth — Firestore cleanup above is still valid
  }

  return NextResponse.json({ ok: true });
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client → DELETE /api/wishlist/[wishlistId] | Untrusted caller sends idToken + wishlistId. Route must verify token and check parentUids/childUid before deleting. |
| client → DELETE /api/auth/user/[uid] | Untrusted caller sends idToken + target uid. Route must verify token and enforce role-based authorization before any deletion. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-01 | Spoofing | Both DELETE routes — idToken | mitigate | `adminAuth.verifyIdToken(idToken)` is called on every request; returns 401 if token invalid, expired, or forged |
| T-07-02 | Elevation of Privilege | DELETE /api/wishlist/[wishlistId] — authorization | mitigate | Checks `data.childUid === decoded.uid` OR `data.parentUids.includes(decoded.uid)` before deletion; returns 403 otherwise |
| T-07-03 | Elevation of Privilege | DELETE /api/auth/user/[uid] child flow — authorization | mitigate | Checks `parentUids.includes(decoded.uid)` before allowing child account deletion; returns 403 if not a parent |
| T-07-04 | Elevation of Privilege | DELETE /api/auth/user/[uid] parent/viewer flow — authorization | mitigate | Checks `decoded.uid === targetUid` (self-only); returns 403 if caller is not the target user |
| T-07-05 | Denial of Service | Both DELETE routes — no rate limiting | accept | Small family app (tens of users). Deletion is a rare, intentional operation. A deleted account cannot be used to repeat the action. Firebase Auth token verification adds inherent throttle. Revisit if app scales. |
| T-07-06 | Tampering | DELETE wishlist — subcollections left orphaned | mitigate | `adminDb.recursiveDelete()` handles all subcollections atomically. Client SDK `deleteDoc()` is NOT used. |
| T-07-07 | Information Disclosure | DELETE /api/auth/user/[uid] — 404 vs 403 ordering | mitigate | Route checks `userSnap.exists` first (404) before authorization check (403). Does not leak whether a UID exists to unauthorized callers beyond the 404/403 distinction, which is acceptable for ASVS L1. |
| T-07-08 | Repudiation | Destructive operations leave no server-side audit trail | accept | Phase 7 scope. Firestore `activityLog` subcollection already handles viewer purchase actions. Account deletion audit logging is a future concern at this scale. |
| T-07-09 | Tampering | Firestore data cleaned but Auth delete fails | mitigate | Firestore is always cleaned first (Task 2 ordering). Auth deletion failure does not create orphaned Firestore data; the worst case is Auth user exists without Firestore data, which is a recoverable state (retry). `auth/user-not-found` is treated as success. |
</threat_model>
